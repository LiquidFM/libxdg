Example1. List all applications able to handle given mime type.

Source code of Example1.

First of all we must include the library header:

#include <xdg/xdg.h>

And "stdio.h" header for printf() function:

#include <stdio.h>

Function print_app() is needed for print information about the given application:

void print_app(const XdgApp *app)
{
const XdgList *values;
const XdgAppGroup *group = xdg_app_group_lookup(app, "Desktop Entry");
if (values = xdg_list_begin(xdg_app_localized_entry_lookup(group, "Name", "ru", "RU", NULL)))
do
printf("Name = %s", xdg_list_item_app_group_entry_value(values));
while (values = xdg_list_next(values));
if (values = xdg_list_begin(xdg_app_localized_entry_lookup(group, "GenericName", "ru", "RU", NULL)))
do
printf("GenericName = %s", xdg_list_item_app_group_entry_value(values));
while (values = xdg_list_next(values));
}

Function print_applications() prints information about all applications able to handle given mime type:

void print_applications(const char *mime)
{
const XdgJointList *apps;

Print information about all user defined applications able to handle given mime type:

if (apps = xdg_joint_list_begin(xdg_apps_lookup(mime)))
do
print_app(xdg_joint_list_item_app(apps));
while (apps = xdg_joint_list_next(apps));
